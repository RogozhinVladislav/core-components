Слайдер. У компонента можно настроить шаг слайдера и подписи для точек.

```tsx live
const min = 1;
const max = 8;
const step = 1;
const steps = ['1', '2', '3', '4', '5', '6', '7', '8'];

const normalizeValue = (value, min, max) =>
    value === '' ? '' : Math.min(max, value.toString().replace(/[^\d-]/g, ''));

render(() => {
    const [value, setValue] = React.useState(1);

    const handleChange = (event, { value }) => setValue(normalizeValue(value, min, max));

    return (
        <>
            <SliderPicker
                value={value}
                min={min}
                max={max}
                step={step}
                onChange={handleChange}
                steps={steps}
            />
            <label style={{marginTop: '20px', display: 'block'}}>Value: {value}</label>
        </>
    );
});
```
